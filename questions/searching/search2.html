        int linearSearch(char arr[], char key, int n) {
            if (arr[n] == '\0')
                return -1;
            
            if (arr[n] == key)
                return n;
        
            return linearSearch(arr, key, n+1);
        }
        void main(){
            char arr[20], key;
            int result;

            printf("enter any string:\n");

            printf("enter value to be searched:\n");
            scanf("%c", &key);
            result = linearSearch(arr, key, 0);
        
            if (result == -1) {
                printf("element %c not found in the array.\n", key);
            } else {
                printf("element %c found at position %d.\n", key, result+1);
