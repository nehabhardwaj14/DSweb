        int binarySearch(int arr[], int n, int key) {
            int l = 0, u = n-1, mid;
            while(l<=u)
            {
                mid = l + (u - l) / 2;
                if (arr[mid] == key)
                {
                    return mid;
                }
                else if(arr[mid] > key){
                    u = mid-1;
                }
                else{
                    l = mid+1;
                }
            }
            return -1; 
        }
