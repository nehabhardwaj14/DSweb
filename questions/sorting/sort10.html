        #include <stdio.h>
        #include<stdlib.h>
        #include <time.h>
        void swap(int* a, int* b) {
            int t = *a;
            *a = *b;
            *b = t;
        }
        
        int partition(int arr[], int low, int high) {
            int pivot = arr[high];
            int i = (low - 1);
        
            for (int j = low; j <= high - 1; j++) {
                if (arr[j] < pivot) {
                    i++;
                    swap(&arr[i], &arr[j]);
                }
            }
        
            swap(&arr[i + 1], &arr[high]);
            return (i + 1);
        }
        
        void quickSort(int arr[], int low, int high) {
            if (low < high) {
                int pi = partition(arr, low, high);
        
                quickSort(arr, low, pi - 1);
                quickSort(arr, pi + 1, high);
            }
        }
        int main()
        {
            int arr[1000], i;
            
            for (i = 0; i < 1000; i++)
            {
                arr[i] = rand(); // Adding 1000 Random Values in array
            }
        
            clock_t start, end;
            double cpu_time_used;
        
            start = clock();
            quickSort(arr, 0, 999);
            end = clock();
            printf("After Sorting Array\n");
            cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
            printf("\nWorst-case Time taken: %f seconds\n", cpu_time_used);
        
            start = clock();
            quickSort(arr, 0, 999);
            end = clock();
            printf("After Sorting Array\n");
            cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
            printf("\nBest-case Time taken: %f seconds\n", cpu_time_used);
            
            return 0;
        }
