      #include<stdio.h>
    #include<stdlib.h>
    #include <time.h>
    void Selectionsort(int *arr, int size){
        int temp, i, j;
        for (i = 0; i < size; i++)
        {
            for (j = i+1; j < size; j++)
            {
                if (arr[i] > arr[j])
                {
                    temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
            
        }
    }
    
    
    int main(){
        int arr[1000], i;
        
        for (i = 0; i < 1000; i++)
        {
            arr[i] = rand(); // Adding 1000 Random Values in array
        }
    
        clock_t start, end;
        double cpu_time_used;
    
        start = clock();
        Selectionsort(arr, 1000);
        end = clock();
        printf("After Sorting Array\n");
        cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
        printf("\nWorst-case Time taken: %f seconds\n", cpu_time_used);
    
        start = clock();
        Selectionsort(arr, 1000);
        end = clock();
        printf("After Sorting Array\n");
        cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
        printf("\nBest-case Time taken: %f seconds\n", cpu_time_used);
    
        return 0;
    }
