        #include <stdio.h>
        #include<stdlib.h>
        #include <time.h>
        void merge(int arr[], int lb, int mid, int ub)
        {
            int beg = lb;
            int beg2 = mid + 1;
            int ind = 0, i, j;
            int merged[1000];
        
            while (beg <= mid && beg2 <= ub)
            {
                if (arr[beg] >= arr[beg2])
                {
                    merged[ind] = arr[beg2++];
                }
                else
                {
                    merged[ind] = arr[beg++];
                }
                ind++;
            }
        
            while (beg <= mid)
            {
                merged[ind++] = arr[beg++];
            }
        
            while (beg2 <= ub)
            {
                merged[ind++] = arr[beg2++];
            }
        
            for (i = 0, j = lb; i < ind; i++, j++)
            {
                arr[j] = merged[i];
            }
        }
        
        void mergeSort(int arr[], int lb, int ub)
        {
            if (lb < ub)
            {
                int mid = lb + (ub - lb) / 2;
                mergeSort(arr, lb, mid);
                mergeSort(arr, mid + 1, ub);
                merge(arr, lb, mid, ub);
            }
        }

        {
            int arr[1000], i;

            clock_t start, end;
            double cpu_time_used;
