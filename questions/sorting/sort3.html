           //bubble sort through iteration
            void bubblesort(int arr[],int n)
              {
               int i,j,temp;
               for(i=0 ;i<n-2;i++){
               for(j=i; j<(n-i-1);j++){
               if(arr[j] >arr[j+1]){
              temp=arr[j];
              arr[j]=arr[j+1];
              arr[j+1]=temp;
              }}}
              }
