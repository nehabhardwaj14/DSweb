PROGRAM: 

        #include<stdio.h>
    #include<stdlib.h>
    #include <time.h>
    void BubbleSort(int *arr, int size){
        int i, j;
        for (i = 0; i < size-2; i++)
        {
            for (j = 0; j < size-i-1; j++)
            {
                if (arr[j] > arr[j+1])
                {
                    int temp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = temp;
                }
            }
        }
    }
    
    
    int main(){
        int arr[1000], i;
        
        for (i = 0; i < 1000; i++)
        {
            arr[i] = rand(); // Adding 1000 Random Values in array
        }
    
        clock_t start, end;
        double cpu_time_used;
    
        start = clock();
        BubbleSort(arr, 1000);
        end = clock();
        printf("After Sorting Array\n");
        cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
        printf("\nWorst-case Time taken: %f seconds\n", cpu_time_used);
    
        start = clock();
        BubbleSort(arr, 1000);
        end = clock();
        printf("After Sorting Array\n");
        cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
        printf("\nBest-case Time taken: %f seconds\n", cpu_time_used);
    
        return 0;
    }
