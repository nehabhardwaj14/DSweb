        #include <stdio.h>
        #include<stdlib.h>
        #include <time.h>
        void InsertionSort(int arr[], int n) {
            if (n <= 1)
                return;
        
            InsertionSort(arr, n - 1);
        
            int key = arr[n - 1];
            int j = n - 2;
        
            while (j >= 0 && arr[j] > key) {
                arr[j + 1] = arr[j];
                j--;
            }
        
            arr[j + 1] = key;
        }
        
        int main() {
            int arr[1000], i;
            
            for (i = 0; i < 1000; i++)
            {
                arr[i] = rand(); // Adding 1000 Random Values in array
            }
        
            clock_t start, end;
            double cpu_time_used;
        
            start = clock();
            InsertionSort(arr, 1000);
            end = clock();
            printf("After Sorting Array\n");
            cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
            printf("\nWorst-case Time taken: %f seconds\n", cpu_time_used);
        
            start = clock();
            InsertionSort(arr, 1000);
            end = clock();
            printf("After Sorting Array\n");
            cpu_time_used =(double) (end-start)/CLOCKS_PER_SEC;
            printf("\nBest-case Time taken: %f seconds\n", cpu_time_used);
            return 0;
        }
