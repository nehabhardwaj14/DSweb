PROGRAM:

        #include <stdio.h>
        #include <string.h>
        #include<conio.h>
        #include<dos.h>
        void displayAlignedText(char text[], int alignment) {
            int length = strlen(text), i;
        
            if (alignment == 1) {
                printf("%s\n", text); 
            } else if (alignment == 2) {
                int spaces = (80 - length) / 2; 
            for (i = 0; i < spaces; i++) {
                printf(" ");
            }
            printf("%s\n", text);
            } else if (alignment == 3) {
            int spaces = 80 - length;
            for (i = 0; i < spaces; i++) {
                printf(" ");
            }
            printf("%s\n", text);
            } else {
            printf("Invalid alignment option.\n");
            }
        }
        
        void main() {
            char text[25];
            int alignment;
            clrscr();
            printf("Enter String:\n");
            gets(text);
            printf("Enter alignment option :\n1 for left\n 2 for center\n 3 for right: \n");
            scanf("%d", &alignment);
        
            displayAlignedText(text, alignment);
        
            delay(5000);
        }
